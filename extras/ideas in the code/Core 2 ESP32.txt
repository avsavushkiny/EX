  for(;;) {
    // Твой код для ядра №2 (APP CPU)
    // Например, обработка того самого "буфера №2"
    
    // Если задача не критична к микросекундам, лучше оставить delay(1)

  }
}

  xTaskCreatePinnedToCore(
    core2Task,   /* Функция задачи */
    "TaskCore2", /* Имя */
    10000,       /* Размер стека */
    NULL,        /* Параметры */
    1,           /* Приоритет */
    NULL,        /* Хендл задачи */
    1            /* ID ядра (0 - PRO, 1 - APP) */

}

Пару советов для твоих буферов:
Приоритет: Если твоя задача по обработке буфера критически важна, ставь приоритет выше (например, 3 или 5).
Volatile: Если ты передаешь флаг готовности буфера между ядрами через глобальную переменную, обязательно помечай её как volatile bool bufferReady;, чтобы компилятор не закешировал значение.
No-RTOS feel: Если хочешь, чтобы задача на ядре 2 работала "как голый код" без прерываний от ОС, можно поднять приоритет до максимума и использовать portDISABLE_INTERRUPTS(), но будь осторожен — это может вызвать ребут по WDT (сторожевому таймеру).
